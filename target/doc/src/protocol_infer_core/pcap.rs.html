1use crate::corpus::{Direction, Flow, UdpDatagram};
2use crate::Error;
3use crate::Result;
4use std::collections::HashMap;
5use std::sync::Arc;
6
7/// Parse un fichier PCAP et extrait les flows UDP
8pub fn parse_pcap(path: &str) -> Result<Vec<Flow>> {
9    use std::fs::File;
10    use std::io::BufReader;
11
12    let file = File::open(path)?;
13    // Utiliser un buffer très grand pour gérer les paquets de grande taille
14    // Certains fichiers PCAP ont des capture lengths de 256KB ou plus
15    let reader = BufReader::with_capacity(262144, file); // 256KB buffer
16    let mut pcap_reader = pcap_parser::create_reader(262144, reader) // 256KB pour le parser
17        .map_err(|e| Error::PcapParse(format!("Failed to create reader: {:?}", e)))?;
18
19    let mut flows: HashMap<(String, String, u16, u16, u8), Vec<UdpDatagram>> = HashMap::new();
20    let mut flow_id_counter = 0usize;
21    let mut flow_id_map: HashMap<(String, String, u16, u16, u8), usize> = HashMap::new();
22
23    loop {
24        match pcap_reader.next() {
25            Ok((offset, pkt)) => {
26                let (ts, data) = match &pkt {
27                    pcap_parser::PcapBlockOwned::LegacyHeader(_) => {
28                        pcap_reader.consume(offset);
29                        continue;
30                    }
31                    pcap_parser::PcapBlockOwned::Legacy(block) => {
32                        let ts = block.ts_sec as f64 + block.ts_usec as f64 / 1_000_000.0;
33                        (ts, &block.data)
34                    }
35                    pcap_parser::PcapBlockOwned::NG(_block) => {
36                        // Support pcapng : pour l'instant, on saute les blocks pcapng
37                        // car l'API est plus complexe et nécessite un parsing spécifique
38                        // TODO: implémenter le support pcapng complet avec extraction des paquets
39                        pcap_reader.consume(offset);
40                        continue;
41                    }
42                };
43
44                // Parser le paquet Ethernet/IP/UDP
45                if let Ok(parsed) = etherparse::PacketHeaders::from_ethernet_slice(data) {
46                    if let Some(ip) = parsed.net {
47                        let (src_ip, dst_ip, ip_header_len) = match &ip {
48                            etherparse::NetHeaders::Ipv4(h, _) => {
49                                use std::net::Ipv4Addr;
50                                (
51                                    Ipv4Addr::from(h.source).to_string(),
52                                    Ipv4Addr::from(h.destination).to_string(),
53                                    h.header_len() as usize,
54                                )
55                            }
56                            etherparse::NetHeaders::Ipv6(_, _) => {
57                                use std::net::Ipv6Addr;
58                                // Pour IPv6, on doit extraire les adresses différemment
59                                // Simplification: utiliser une valeur par défaut
60                                ("::1".to_string(), "::1".to_string(), 40)
61                            }
62                            _ => continue,
63                        };
64
65                        if let Some(udp) = parsed.transport {
66                            if let etherparse::TransportHeader::Udp(udp_header) = udp {
67                                let src_port = udp_header.source_port;
68                                let dst_port = udp_header.destination_port;
69
70                                // Calculer l'offset du payload
71                                let udp_header_len = 8;
72                                let eth_header_len = 14; // Ethernet header
73                                let payload_start = eth_header_len + ip_header_len + udp_header_len;
74                                
75                                let payload = if payload_start < data.len() {
76                                    Arc::from(&data[payload_start..])
77                                } else {
78                                    continue;
79                                };
80
81                                let five_tuple = (
82                                    src_ip.clone(),
83                                    dst_ip.clone(),
84                                    src_port,
85                                    dst_port,
86                                    17u8, // UDP
87                                );
88
89                                let flow_id = *flow_id_map
90                                    .entry(five_tuple.clone())
91                                    .or_insert_with(|| {
92                                        flow_id_counter += 1;
93                                        flow_id_counter - 1
94                                    });
95
96                                let direction = if flow_id % 2 == 0 {
97                                    Direction::ClientToServer
98                                } else {
99                                    Direction::ServerToClient
100                                };
101
102                                let datagram = UdpDatagram {
103                                    timestamp: ts,
104                                    flow_id,
105                                    direction,
106                                    payload,
107                                };
108
109                                flows.entry(five_tuple).or_default().push(datagram);
110                            }
111                        }
112                    }
113                }
114
115                pcap_reader.consume(offset);
116            }
117            Err(pcap_parser::PcapError::Eof) => break,
118            Err(pcap_parser::PcapError::Incomplete(needed)) => {
119                // Erreur Incomplete : le parser a besoin de plus de données
120                // Cela peut arriver avec des paquets très grands ou des fichiers corrompus
121                return Err(Error::PcapParse(format!(
122                    "PCAP parsing error: Incomplete data (need {} more bytes). \
123                    Le fichier peut être corrompu, incomplet, ou contenir des paquets très grands. \
124                    Essayez d'augmenter la taille du buffer ou vérifiez l'intégrité du fichier.",
125                    needed
126                )));
127            }
128            Err(e) => {
129                return Err(Error::PcapParse(format!("PCAP parsing error: {:?}", e)));
130            }
131        }
132    }
133
134    let mut result: Vec<Flow> = flows
135        .into_iter()
136        .map(|((src_ip, dst_ip, src_port, dst_port, protocol), datagrams)| Flow {
137            src_ip,
138            dst_ip,
139            src_port,
140            dst_port,
141            protocol,
142            datagrams,
143        })
144        .collect();
145
146    result.sort_by_key(|f| f.datagrams.len());
147    result.reverse();
148
149    Ok(result)
150}
