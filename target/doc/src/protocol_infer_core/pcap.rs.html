1use crate::corpus::{Direction, Flow, UdpDatagram};
2use crate::Error;
3use crate::Result;
4use std::collections::HashMap;
5use std::sync::Arc;
6
7/// Parse un fichier PCAP et extrait tous les paquets UDP dans un seul flow
8pub fn parse_pcap(path: &str) -> Result<Flow> {
9    use std::fs::File;
10    use std::io::BufReader;
11
12    let file = File::open(path)?;
13    // Buffer généreux pour couvrir des paquets/snaplen très grands (snaplen vu à 262144)
14    // On prend 1 MiB pour limiter les risques d'Incomplete répétés.
15    let reader = BufReader::with_capacity(1_048_576, file);
16    let mut pcap_reader = pcap_parser::create_reader(1_048_576, reader)
17        .map_err(|e| Error::PcapParse(format!("Failed to create reader: {:?}", e)))?;
18
19    let mut all_datagrams: Vec<UdpDatagram> = Vec::new();
20
21    loop {
22        match pcap_reader.next() {
23            Ok((offset, pkt)) => {
24                let (ts, data) = match &pkt {
25                    pcap_parser::PcapBlockOwned::LegacyHeader(_) => {
26                        pcap_reader.consume(offset);
27                        continue;
28                    }
29                    pcap_parser::PcapBlockOwned::Legacy(block) => {
30                        let ts = block.ts_sec as f64 + block.ts_usec as f64 / 1_000_000.0;
31                        (ts, &block.data)
32                    }
33                    pcap_parser::PcapBlockOwned::NG(_block) => {
34                        // Support pcapng : non géré pour l'instant
35                        pcap_reader.consume(offset);
36                        continue;
37                    }
38                };
39
40                // Parser le paquet Ethernet/IP/UDP
41                if let Ok(parsed) = etherparse::PacketHeaders::from_ethernet_slice(data) {
42                    if let Some(ip) = parsed.net {
43                        let (src_ip, dst_ip, ip_header_len, ip_total_length, is_fragmented) = match &ip {
44                            etherparse::NetHeaders::Ipv4(h, _) => {
45                                use std::net::Ipv4Addr;
46                                // Vérifier la fragmentation : fragment_offset = 0 et More Fragments = 0
47                                let fragment_offset = h.fragment_offset();
48                                let more_fragments = h.more_fragments();
49                                let is_fragmented = fragment_offset != 0 || more_fragments;
50                                
51                                (
52                                    Ipv4Addr::from(h.source).to_string(),
53                                    Ipv4Addr::from(h.destination).to_string(),
54                                    h.header_len() as usize,
55                                    h.total_len() as usize,
56                                    is_fragmented,
57                                )
58                            }
59                            etherparse::NetHeaders::Ipv6(_, _) => {
60                                // IPv6 : utiliser la longueur du payload IPv6
61                                // Pour IPv6, on utilise la longueur du paquet complet moins les headers
62                                ("::1".to_string(), "::1".to_string(), 40, data.len(), false)
63                            }
64                            _ => continue,
65                        };
66
67                        if let Some(udp) = parsed.transport {
68                            if let etherparse::TransportHeader::Udp(udp_header) = udp {
69                                let src_port = udp_header.source_port;
70                                let dst_port = udp_header.destination_port;
71
72                                // Calculer l'offset du payload UDP
73                                let udp_header_len = 8;
74                                let eth_header_len = 14; // Ethernet header
75                                let payload_start = eth_header_len + ip_header_len + udp_header_len;
76
77                                // Calculer la taille réelle du payload UDP en utilisant la longueur IP
78                                // IP total length = IP header + UDP header + UDP payload
79                                // UDP payload length = IP total length - IP header - UDP header
80                                let udp_payload_length = if !is_fragmented && ip_total_length >= ip_header_len + udp_header_len {
81                                    ip_total_length - ip_header_len - udp_header_len
82                                } else {
83                                    // Si fragmenté ou longueur invalide, utiliser la taille disponible
84                                    if payload_start < data.len() {
85                                        data.len() - payload_start
86                                    } else {
87                                        pcap_reader.consume(offset);
88                                        continue;
89                                    }
90                                };
91
92                                // Extraire le payload UDP avec la taille correcte
93                                let payload = if payload_start + udp_payload_length <= data.len() {
94                                    Arc::from(&data[payload_start..payload_start + udp_payload_length])
95                                } else if payload_start < data.len() {
96                                    // Si la taille calculée dépasse les données disponibles,
97                                    // utiliser ce qui est disponible (cas de capture tronquée)
98                                    Arc::from(&data[payload_start..])
99                                } else {
100                                    pcap_reader.consume(offset);
101                                    continue;
102                                };
103
104                                // Tous les paquets sont dans le même flow (flow_id = 0)
105                                let flow_id = 0;
106                                
107                                // Direction basée sur l'ordre des paquets (alternance simple)
108                                let direction = if all_datagrams.len() % 2 == 0 {
109                                    Direction::ClientToServer
110                                } else {
111                                    Direction::ServerToClient
112                                };
113
114                                let datagram = UdpDatagram {
115                                    timestamp: ts,
116                                    flow_id,
117                                    direction,
118                                    payload,
119                                };
120
121                                all_datagrams.push(datagram);
122                            }
123                        }
124                    }
125                }
126
127                // Réinitialiser le compteur d'Incomplete après un succès
128                // (plus utilisé, laissé pour compat éventuelle)
129                pcap_reader.consume(offset);
130            }
131            Err(pcap_parser::PcapError::Eof) => break,
132            Err(pcap_parser::PcapError::Incomplete(_needed)) => {
133                // Re-remplir le buffer et réessayer
134                pcap_reader
135                    .refill()
136                    .map_err(|e| Error::PcapParse(format!("PCAP refill error: {:?}", e)))?;
137                continue;
138            }
139            Err(e) => {
140                return Err(Error::PcapParse(format!("PCAP parsing error: {:?}", e)));
141            }
142        }
143    }
144
145    // Créer un seul flow avec tous les paquets
146    // Utiliser les valeurs du premier paquet pour les métadonnées du flow
147    let flow = if all_datagrams.is_empty() {
148        Flow {
149            src_ip: "0.0.0.0".to_string(),
150            dst_ip: "0.0.0.0".to_string(),
151            src_port: 0,
152            dst_port: 0,
153            protocol: 17, // UDP
154            datagrams: Vec::new(),
155        }
156    } else {
157        // Trier par timestamp pour avoir un ordre chronologique
158        all_datagrams.sort_by(|a, b| a.timestamp.partial_cmp(&b.timestamp).unwrap_or(std::cmp::Ordering::Equal));
159        
160        Flow {
161            src_ip: "All".to_string(),
162            dst_ip: "All".to_string(),
163            src_port: 0,
164            dst_port: 0,
165            protocol: 17, // UDP
166            datagrams: all_datagrams,
167        }
168    };
169
170    Ok(flow)
171}
